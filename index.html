def lin_reg(df):
    x_years = df['year']
    y_lifeExp = df['lifeExp']

    x_years = x_years.values.reshape((df.shape[0],1))
    y_lifeExp = y_lifeExp.values.reshape((df.shape[0],1))

    # Create linear regression object
    regr = linear_model.LinearRegression()

    # Train the model using the training sets
    regr.fit(x_years, y_lifeExp)

    # Make predictions using the testing set
    y_lifeExp_pred = regr.predict(x_years)

    y_lifeExp_pred = np.concatenate(y_lifeExp_pred,axis=0)
    x_years = np.concatenate(x_years,axis=0)

    # # Plot outputs
    plt.plot(x_years, y_lifeExp_pred)

    print('Coefficient: ', regr.coef_[0][0])
    
    return y_lifeExp_pred

def logistic_grad_descent(X, y, T, alpha):
    m, n = X.shape
    theta = np.zeros(n)
    f = np.zeros(T)
    for i in range(T):
        p = np.exp(X.dot(theta))/(1+np.exp(X.dot(theta)))
        f[i] = ((1/np.log(2))*np.log(1+np.exp(-(y*p)))).sum()
        g = (p-y).dot(X)
        theta = theta - alpha*g
    return theta, f

def linear_grad_descent(X, y, T, alpha):
    m, n = X.shape
    theta = np.zeros(n)
    f = np.zeros(T)
    for i in range(T):
        f[i] = 0.5*np.linalg.norm(X.dot(theta)-y)**2
        g = X.T.dot(X.dot(theta)-y)
        theta = theta - alpha*g
    return theta,f
